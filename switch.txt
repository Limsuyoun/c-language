#include<stdio.h>
int main() {
	int n, data[1000], i, m, data1[100], data2[100], j, a = 1;
	data[0] = 34;
	scanf("%d", &n);

	for (i = 1; i <= n; i++) {
		scanf("%d", &data[i]);
	}

	scanf("%d", &m);

	for (i = 1; i <= m; i++) {
		scanf("%d", &data1[i]);
		scanf("%d", &data2[i]);
	}

	for (i = 1; i <= m; i++) {
		if (data1[i] == 1) {  //³²ÀÚ 
			for (j = 1; j <= n; j++) {
				if (j % data2[i] == 0) {
					if (data[j] == 1)
						data[j] = 0;
					else
						data[j] = 1;
				}
			}
		}
		else {
			if (data[data2[i]] == 1)
				data[data2[i]] = 0;
			else
				data[data2[i]] = 1;
			if (data2[i] != 1 && data2[i] != n){
				for (j = 1; j <= n; j++) {
					if (data[data2[i] - a] == data[data2[i] + a] && data2[i] - a >= 1 && data2[i] + a <= n) {
						if (data[data2[i] - a] == 0)
							data[data2[i] - a] = 1;
						else
							data[data2[i] - a] = 0;

						if (data[data2[i] + a] == 0)
							data[data2[i] + a] = 1;
						else
							data[data2[i] + a] = 0;
						a++;
					}
					else
						break;
			     	}
				}
			}
		a = 1;
	}

	for (i = 1; i <= n; i++) {
		//if (i == 1 && i % 20 == 1)

			printf("%d ", data[i]);

	}
}
