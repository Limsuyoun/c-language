#include<stdio.h>
int main() {
	int n,i,j,m=1;
	scanf("%d", &n);
	for (i = 0; i < n; i++) {
		for (j = 0; j < n; j++) {
			if (i == 0 || i == n - 1) {
				if (m == 1) {
					printf("*");
					m = 0;
				}
				else {
					printf(" ");
					m = 1;
				}
			}
			else {
				if (j == n / 2)
					printf("*");
				else
					printf(" ");
			}
		}
		printf("\n");
		m = 1;
	}
}
